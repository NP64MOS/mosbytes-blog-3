## Building the Basic Chatbot

Let's start with a simple rule-based chatbot:

```python
import re
import random

class SimpleChatbot:
    def __init__(self):
        self.patterns = {
            r'hello|hi|hey': [
                "Hello! How can I help you today?",
                "Hi there! What's on your mind?",
                "Hey! Nice to meet you!"
            ],
            r'how are you': [
                "I'm doing great, thanks for asking!",
                "I'm here and ready to help!",
                "Fantastic! How about you?"
            ],
            r'what.*your.*name': [
                "I'm your friendly AI assistant!",
                "You can call me ChatBot!",
                "I'm an AI created to help you!"
            ],
            r'bye|goodbye|see you': [
                "Goodbye! Have a great day!",
                "See you later!",
                "Bye! Come back anytime!"
            ]
        }
    
    def get_response(self, user_input):
        user_input = user_input.lower()
        
        for pattern, responses in self.patterns.items():
            if re.search(pattern, user_input):
                return random.choice(responses)
        
        return "I'm not sure how to respond to that. Can you try rephrasing?"

# Test the chatbot
bot = SimpleChatbot()
print(bot.get_response("Hello there!"))
```

## Adding Natural Language Processing

Now let's make our chatbot smarter using NLTK:

```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer

# Download required NLTK data
nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')

class SmartChatbot:
    def __init__(self):
        self.lemmatizer = WordNetLemmatizer()
        self.stop_words = set(stopwords.words('english'))
        
        # Knowledge base
        self.knowledge = {
            'ai': "AI stands for Artificial Intelligence. It's technology that enables machines to simulate human intelligence.",
            'python': "Python is a popular programming language, especially for AI and data science.",
            'machine learning': "Machine learning is a subset of AI that enables computers to learn from data.",
            'chatbot': "A chatbot is an AI program designed to simulate conversation with humans."
        }
    
    def preprocess_text(self, text):
        # Tokenize and convert to lowercase
        tokens = word_tokenize(text.lower())
        
        # Remove stopwords and lemmatize
        processed_tokens = [
            self.lemmatizer.lemmatize(token) 
            for token in tokens 
            if token not in self.stop_words and token.isalpha()
        ]
        
        return processed_tokens
    
    def find_best_match(self, user_input):
        user_tokens = self.preprocess_text(user_input)
        
        best_match = None
        best_score = 0
        
        for topic, response in self.knowledge.items():
            topic_tokens = self.preprocess_text(topic)
            
            # Calculate similarity (simple word overlap)
            common_words = set(user_tokens) & set(topic_tokens)
            score = len(common_words) / len(topic_tokens) if topic_tokens else 0
            
            if score > best_score and score > 0.5:
                best_score = score
                best_match = response
        
        return best_match
    
    def get_response(self, user_input):
        # Check knowledge base first
        knowledge_response = self.find_best_match(user_input)
        if knowledge_response:
            return knowledge_response
        
        # Fallback to simple patterns
        simple_bot = SimpleChatbot()
        return simple_bot.get_response(user_input)

# Test the smart chatbot
smart_bot = SmartChatbot()
print(smart_bot.get_response("What is artificial intelligence?"))
```

## Adding Personality and Context

Let's make our chatbot more engaging:

```python
class PersonalizedChatbot(SmartChatbot):
    def __init__(self):
        super().__init__()
        self.user_name = None
        self.conversation_history = []
        
        # Personality traits
        self.personality = {
            'greeting': "I'm excited to chat with you!",
            'encouragement': "That's a great question!",
            'curiosity': "Tell me more about that!"
        }
    
    def remember_name(self, user_input):
        # Simple name extraction
        if "my name is" in user_input.lower():
            parts = user_input.lower().split("my name is")
            if len(parts) > 1:
                self.user_name = parts[1].strip().title()
                return f"Nice to meet you, {self.user_name}! I'll remember that."
        return None
    
    def get_response(self, user_input):
        # Remember conversation
        self.conversation_history.append(user_input)
        
        # Check for name
        name_response = self.remember_name(user_input)
        if name_response:
            return name_response
        
        # Personalized greeting
        if self.user_name and any(word in user_input.lower() for word in ['hello', 'hi', 'hey']):
            return f"Hello again, {self.user_name}! {self.personality['greeting']}"
        
        # Get regular response
        response = super().get_response(user_input)
        
        # Add personality occasionally
        if random.random() < 0.3:  # 30% chance
            response += f" {random.choice(list(self.personality.values()))}"
        
        return response

# Test personalized chatbot
personal_bot = PersonalizedChatbot()
print(personal_bot.get_response("Hi, my name is Alice"))
print(personal_bot.get_response("Hello"))
```
